引入 ​

js
import Vue from "vue";
import VueI18n from "vue-i18n";
import zhCN from "./lang/zh-CN.json";
import en from "./lang/en.json";
import { getSaveLangauge } from "@/utils/i18n";

Vue.use(VueI18n);

const messages = {
  "zh-CN": zhCN,
  en
};

const i18n = new VueI18n({
  locale: getSaveLangauge(),
  fallbackLocale: "zh-CN",
  messages,
  silentFallbackWarn: true, // 不显示 fallback 警告
  silentTranslationWarn: true // 不显示 missing translation 警告
});

export default i18n;

工具类及使用 ​

js
import i18n from "@/i18n";
import Utils from "./utils";
/**
 * 设置语言
 * @param {*} locale - 语言标识
 * @returns
 */
export function setLanguage(locale) {
  if (!locale || i18n.locale === locale) return;

  i18n.locale = locale;
  try {
    uni.setStorageSync("locale", locale);
  } catch (e) {
    console.warn("语言设置保存失败", e);
  }
}

/**
 * 切换底部栏语言
 */
export function changeTabBarLangauge() {
  if (typeof __uniConfig === "undefined" || !__uniConfig.tabBar) {
    return;
  }

  const pages = getCurrentPages();
  const currentPage = pages[pages.length - 1];
  const currentRoute = currentPage.route;

  const tabBarPages = __uniConfig.tabBar.list.map(item => item.pagePath);

  if (tabBarPages.includes(currentRoute)) {
    uni.setTabBarItem({ index: 0, text: i18n.t("首页") });
    uni.setTabBarItem({ index: 1, text: i18n.t("相册") });
    uni.setTabBarItem({ index: 2, text: i18n.t("美化") });
    uni.setTabBarItem({ index: 3, text: i18n.t("我的") });
  } else {
    console.log("跳过 tabBar 设置，当前页面不是 tab 页面");
  }
}

/**
 * 获取保存的语言设置
 * @returns - 返回语言标识
 */
export function getSaveLangauge() {
  const defaultLanguage = "zh-CN";
  try {
    return Utils.isEmpty(uni.getStorageSync("locale")) ? defaultLanguage : uni.getStorageSync("locale");
  } catch {
    return defaultLanguage;
  }
}

/**
 * 获取当前使用语言
 * @returns - 返回语言标识
 */
export function getCurrentLangauge() {
  return i18n.locale;
}

/**
 * 获取 i18n 翻译内容
 * @param {String} key - 需要翻译的标识
 * @returns {String} - 翻译后的文本
 */
export function translate(key) {
  return i18n.t(key);
}
